<script lang="ts">
	// Dream bubble generator
	function createDreamBubble(event: MouseEvent | TouchEvent) {
		const container = document.querySelector('.dream-bubble-container');
		if (!container) return;

		const bubble = document.createElement('div');
		bubble.classList.add('dream-bubble');

		// Random size between 30px and 80px
		const size = 30 + Math.random() * 50;
		bubble.style.width = `${size}px`;
		bubble.style.height = `${size}px`;

		// Set position to click coordinates
		const clientX =
			'touches' in event && event.touches[0]
				? event.touches[0].clientX
				: 'clientX' in event
					? event.clientX
					: 0;
		const clientY =
			'touches' in event && event.touches[0]
				? event.touches[0].clientY
				: 'clientY' in event
					? event.clientY
					: 0;

		bubble.style.left = `${clientX - size / 2}px`;
		bubble.style.top = `${clientY - size / 2}px`;

		// Random movement direction
		bubble.style.setProperty('--rand-x', (Math.random() * 2 - 1).toString());
		bubble.style.setProperty('--rand-y', (Math.random() * 1 + 0.5).toString());

		container.appendChild(bubble);

		// Remove bubble after animation completes
		setTimeout(() => {
			bubble.remove();
		}, 8000);
	}

	// Slider drag functionality
	if (typeof window !== 'undefined') {
		window.addEventListener('DOMContentLoaded', () => {
			// Initialize dream bubbles
			const bubbleContainer = document.createElement('div');
			bubbleContainer.classList.add('dream-bubble-container');
			document.body.appendChild(bubbleContainer);

			// Add click listener to document for dream bubbles
			document.addEventListener('click', (e: MouseEvent) => createDreamBubble(e));
			document.addEventListener('touchstart', (e: TouchEvent) => createDreamBubble(e));

			// Initialize tide timer
			updateTideTimer();
			setInterval(updateTideTimer, 60000);

			// Initialize before-after slider
			initBeforeAfterSlider();

			// Initialize ambient sound controls
			initAmbientSounds();
		});
	}

	// Tide timer functionality
	function updateTideTimer() {
		const tideElement = document.querySelector('.tide-timer');
		if (!tideElement) return;

		const now = new Date();
		const hour = now.getHours();

		// Determine tide status based on time of day
		// Morning: Rising tide
		// Afternoon: High tide
		// Evening: Falling tide
		// Night: Low tide
		let tideStatus = '';
		let tideIcon = '';
		let tideColor = '';

		if (hour >= 6 && hour < 12) {
			tideStatus = 'Rising Inspiration';
			tideIcon = 'ðŸŒŠ';
			tideColor = 'from-blue-400 to-cyan-300';
		} else if (hour >= 12 && hour < 18) {
			tideStatus = 'Peak Creativity';
			tideIcon = 'ðŸŒŠðŸŒŠ';
			tideColor = 'from-blue-500 to-teal-400';
		} else if (hour >= 18 && hour < 22) {
			tideStatus = 'Flowing Ideas';
			tideIcon = 'ðŸŒŠ';
			tideColor = 'from-blue-600 to-teal-500';
		} else {
			tideStatus = 'Dream Waters';
			tideIcon = 'ðŸŒ™';
			tideColor = 'from-blue-800 to-indigo-600';
		}

		tideElement.innerHTML = `
			<div class="flex items-center gap-2">
				<div class="tide-icon">${tideIcon}</div>
				<div>
					<div class="text-xs text-muted-foreground">Current Tide</div>
					<div class="font-medium bg-gradient-to-r ${tideColor} bg-clip-text text-transparent">${tideStatus}</div>
				</div>
			</div>
		`;
	}

	// Before-after slider functionality
	function initBeforeAfterSlider() {
		const slider = document.querySelector('.before-after-slider');
		if (!slider) return;

		const handle = slider.querySelector('.handle-circle');
		const afterImage = slider.querySelector('.after-image');
		const sliderContainer = slider.querySelector('.slider-container');

		if (!handle || !afterImage || !sliderContainer) return;

		let isDragging = false;

		// Update image clipping based on handle position
		function updateClip(x: number) {
			const containerWidth = sliderContainer!.getBoundingClientRect().width;
			const percentage = (x / containerWidth) * 100;
			(afterImage as HTMLElement).style.clipPath =
				`polygon(${percentage}% 0, 100% 0, 100% 100%, ${percentage}% 100%)`;
			(handle as HTMLElement).style.left = `${percentage}%`;
		}

		// Handle mouse events
		handle.addEventListener('mousedown', () => {
			isDragging = true;
		});

		document.addEventListener('mousemove', (e: MouseEvent) => {
			if (!isDragging) return;

			const rect = sliderContainer!.getBoundingClientRect();
			const x = Math.max(0, Math.min(e.clientX - rect.left, rect.width));
			updateClip(x);

			e.preventDefault();
		});

		document.addEventListener('mouseup', () => {
			isDragging = false;
		});

		// Handle touch events for mobile
		handle.addEventListener('touchstart', (e: Event) => {
			isDragging = true;
			e.preventDefault();
		});

		document.addEventListener('touchmove', (e: Event) => {
			if (!isDragging) return;

			// Type assertion for TouchEvent
			const touchEvent = e as TouchEvent;
			if (!touchEvent.touches || !touchEvent.touches[0]) return;

			const rect = sliderContainer!.getBoundingClientRect();
			const x = Math.max(0, Math.min(touchEvent.touches[0].clientX - rect.left, rect.width));
			updateClip(x);

			e.preventDefault();
		});

		document.addEventListener('touchend', () => {
			isDragging = false;
		});
	}

	// Ambient ocean sounds
	function initAmbientSounds() {
		const soundToggle = document.querySelector('.sound-toggle');
		if (!soundToggle) return;

		let audio: HTMLAudioElement | null = null;
		let isPlaying = false;

		soundToggle.addEventListener('click', () => {
			if (!audio) {
				// Create audio element for ocean sounds
				audio = new Audio('https://assets.mixkit.co/sfx/preview/mixkit-sea-waves-loop-1196.mp3');
				audio.loop = true;
				audio.volume = 0.2;
			}

			if (isPlaying) {
				audio.pause();
				isPlaying = false;
				soundToggle.classList.remove('sound-on');
				soundToggle.classList.add('sound-off');
				(soundToggle as HTMLElement).setAttribute('title', 'Enable ambient ocean sounds');
			} else {
				audio.play().catch((e) => console.error('Could not play audio:', e));
				isPlaying = true;
				soundToggle.classList.remove('sound-off');
				soundToggle.classList.add('sound-on');
				(soundToggle as HTMLElement).setAttribute('title', 'Disable ambient ocean sounds');
			}
		});
	}

	// Helper function to get the right icon component
	function getIconComponent(iconName: string) {
		switch (iconName) {
			case 'Zap':
				return Zap;
			case 'Image':
				return Image;
			case 'MousePointer':
				return MousePointer;
			case 'Waves':
				return Waves;

				return CloudSun;
			default:
				return Sparkles;
		}
	}

	// Wave animation variables
	let mouseX = 0;
	let mouseY = 0;
	let waveCanvas: HTMLCanvasElement;
	let waveCtx: CanvasRenderingContext2D | null;
	let wavePoints: { x: number; y: number; originalY: number; speed: number }[] = [];
	let canvasWidth = 0;
	let canvasHeight = 0;
	let animationFrame: number;
	let soundEnabled = false;
	let audio: HTMLAudioElement;

	// Dream bubble container
	let dreamBubbleContainer: HTMLDivElement;

	function handleMouseMove(event: MouseEvent) {
		mouseX = event.clientX;
		mouseY = event.clientY;

		// Create wave effect at mouse position if within wave canvas area
		if (waveCtx && mouseY > 0 && mouseY < canvasHeight) {
			const waveIndex = Math.floor((mouseX / canvasWidth) * wavePoints.length);
			if (wavePoints[waveIndex]) {
				wavePoints[waveIndex].y = mouseY * 0.5;
			}
		}
	}

	function initInteractiveWave() {
		if (!waveCanvas) return;

		waveCtx = waveCanvas.getContext('2d');
		if (!waveCtx) return;

		// Set canvas size
		canvasWidth = window.innerWidth;
		canvasHeight = 150;
		waveCanvas.width = canvasWidth;
		waveCanvas.height = canvasHeight;

		// Create wave points
		const totalPoints = Math.floor(canvasWidth / 20);
		wavePoints = [];

		for (let i = 0; i < totalPoints; i++) {
			const x = (canvasWidth / totalPoints) * i;
			const y = canvasHeight * 0.5;
			wavePoints.push({
				x,
				y,
				originalY: y,
				speed: 0.05 + Math.random() * 0.05
			});
		}

		// Start animation
		animateWave();
	}

	function animateWave() {
		if (!waveCtx) return;

		// Clear canvas
		waveCtx.clearRect(0, 0, canvasWidth, canvasHeight);

		// Draw wave
		waveCtx.beginPath();
		waveCtx.moveTo(0, canvasHeight);

		// Update wave points
		for (let i = 0; i < wavePoints.length; i++) {
			const point = wavePoints[i];

			// Animate points back to original position
			point.y += (point.originalY - point.y) * point.speed;

			// Draw curve between points
			if (i === 0) {
				waveCtx.lineTo(point.x, point.y);
			} else {
				const prevPoint = wavePoints[i - 1];
				const xc = (prevPoint.x + point.x) / 2;
				const yc = (prevPoint.y + point.y) / 2;
				waveCtx.quadraticCurveTo(prevPoint.x, prevPoint.y, xc, yc);
			}
		}

		// Complete the wave path
		waveCtx.lineTo(canvasWidth, canvasHeight);
		waveCtx.lineTo(0, canvasHeight);
		waveCtx.closePath();

		// Fill with gradient
		const gradient = waveCtx.createLinearGradient(0, 0, canvasWidth, 0);
		gradient.addColorStop(0, 'rgba(59, 130, 246, 0.4)');
		gradient.addColorStop(0.5, 'rgba(45, 212, 191, 0.4)');
		gradient.addColorStop(1, 'rgba(59, 130, 246, 0.4)');
		waveCtx.fillStyle = gradient;
		waveCtx.fill();

		// Continue animation
		animationFrame = requestAnimationFrame(animateWave);
	}

	onMount(() => {
		initInteractiveWave();

		// Handle window resize
		const handleResize = () => {
			if (animationFrame) {
				cancelAnimationFrame(animationFrame);
			}
			initInteractiveWave();
		};

		window.addEventListener('resize', handleResize);

		return () => {
			window.removeEventListener('resize', handleResize);
			if (animationFrame) {
				cancelAnimationFrame(animationFrame);
			}
		};
	});
</script>
